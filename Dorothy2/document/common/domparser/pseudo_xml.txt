DOMParser/XML pseudo version


Usage
=====

  var text = '<foo><bar /></foo>';
  var type = 'text/xml';

  common_load ('domparser\xml');

  var domparser = new DOMParser ();
  var document;

  try {
    document = domparser.parseFromString (text, type);

    var e = document.getElementsByTagName ('bar');
  }
  finally {
    if (document) {
      document.destroy ();
      document = null;
    }
  }
